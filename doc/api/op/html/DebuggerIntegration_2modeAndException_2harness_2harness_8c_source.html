79 //
80 // Apply a watchpoint on a processor memory location
81 //
82 static void applyWatchpoint(
83  optProcessorP proc,
84  Addr address,
85  Addr size,
86  watchType type
87 ) {
88 
89  optWatchpointP wp;
90  const char *typeName = "???";
91 
92  switch(type) {
93 
94  case (WATCH_RW):
95  // create access watchpoint
96  wp = opProcessorAccessWatchpointNew(
97  proc, False, address, address+size-1, (void *)(id++), 0
98  );
99  typeName = "ACCESS";
100  break;
101 
102  case (WATCH_READ):
103  // create read watchpoint
104  wp = opProcessorReadWatchpointNew(
105  proc, False, address, address+size-1, (void *)(id++), 0
106  );
107  typeName = "READ";
108  break;
109 
110  case (WATCH_WRITE):
111  wp = opProcessorWriteWatchpointNew(
112  proc, False, address, address+size-1, (void *)(id++), 0
113  );
114  typeName = "WRITE";
115  break;
116 
117  case (WATCH_MODE):
118  wp = opProcessorModeWatchpointNew(
119  proc, (void *)(id++), 0
120  );
121  typeName = "MODE";
122  break;
123 
124  case (WATCH_EXCEPTION):
125  wp = opProcessorExceptionWatchpointNew(
126  proc, (void *)(id++), 0
127  );
128  typeName = "EXCEPTION";
129  break;
130 
131  default:
132  opPrintf("Invalid watchpoint type %d requested\n", type);
133  return;
134 
135  }
136 
137  opPrintf("%s: %s watchpoint is %u\n", opObjectName(proc), typeName, getWatchpointId(wp));
138 
139 }

141 //
142 // Handle all active watchpoints
143 //
144 static void handleWatchpoints(optModuleP mi) {
145 
146  optWatchpointP wp;
147 
148  while((wp=opRootModuleWatchpointNext(mi))) {
149 
150  Uns32 id = getWatchpointId(wp);
151  optProcessorP processor = opWatchpointTriggeredBy(wp);
152  optWatchpointType type = opWatchpointType(wp);
153 
154  switch(type) {
155 
156  case OP_WP_MODE: {
157 
158  // a mode switch watchpoint was triggered
159  optModeP mode = opProcessorModeCurrent(processor);
160  Uns32 modeCode = mode ? opModeCode(mode) : 0;
161  const char *modeName = mode ? opModeName(mode) : "none";
162 
163  // report new mode
164  opPrintf(
165  " watchpoint %u (processor %s:MODE) triggered mode -> %d (%s)\n",
166  id,
167  opObjectName(processor),
168  modeCode,
169  modeName ?: ""
170  );
171 
172  opWatchpointReset(wp);
173 
174  break;
175  }
176 
177  case OP_WP_EXCEPTION: {
178 
179  // an exception watchpoint was triggered
180  optExceptionP except = opProcessorExceptionCurrent(processor);
181  Uns32 exceptCode = except ? opExceptionCode(except) : 0;
182  const char *exceptName = except ? opExceptionName(except) : "none";
183  const char *exceptDesc = except ? opExceptionDescription(except) : "";
184 
185  // report current exception
186  opPrintf(
187  " watchpoint %u (processor %s:EXCEPTION) triggered exception -> %d %s: %s)\n",
188  id,
189  opObjectName(processor),
190  exceptCode,
191  exceptName ?: "",
192  exceptDesc ?: ""
193  );
194 
195  opWatchpointReset(wp);
196 
197  break;
198  }
199 
200  default: {
201 
202  opPrintf(
203  " unknown watchpoint type %u triggered by processor %s\n",
204  type,
205  opObjectName (processor)
206  );
207 
208  opWatchpointReset(wp);
209 
210  break;
211  }
212  }
213  }
214 }

216 //
217 // Main routine
218 //
219 int main(int argc, const char *argv[]) {
220 
221  // Start up simulator and parse command line arguments
222  opSessionInit(OP_VERSION);
223  opCmdParseStd (argv[0], OP_AC_ALL, argc, argv);
224 
225  // Load the design
226  optModuleP root = opRootModuleNew(0, 0, 0);
227  opModuleNew(root, MODULE_DIR, MODULE_INSTANCE, 0, 0);
228 
229  // Get processor handles
230  optProcessorP processor = opObjectByName(root, MODULE_INSTANCE "/" CPU_INSTANCE, OP_PROCESSOR_EN).Processor;
231 
232  // Load application programs
233  opProcessorApplicationLoad(processor, APP_DIR "/" APP_ASMTEST, APP_LDR_CTRL, 0);
234 
235  // Must advance simulator phase before setting watchpoints
236  opRootModulePreSimulate(root);
237 
238  // Apply watchpoints on mode switches and exceptions
239  applyWatchpoint(processor, 0, 0, WATCH_MODE);
240  applyWatchpoint(processor, 0, 0, WATCH_EXCEPTION);
241 
242  Bool stepOver = False;
243  Bool finished = False;
244  optProcessorP stopProcessor = 0;
245 
246  // Simulation loop
247  while(!finished) {
248 
249  // Simulate the platform using the default scheduler
250  // All breakpoints are set and cleared together, ala gdb behavior
251  if(stepOver) {
252  opProcessorBreakpointICountSet(stopProcessor, 1);
253  stopProcessor = opRootModuleSimulate (root);
254  stepOver = False;
255  } else {
256  applyBreakpoints(processor, breakpoints);
257  stopProcessor = opRootModuleSimulate(root);
258  clearBreakpoints(processor, breakpoints);
259  }
260 
261  optStopReason sr = stopProcessor ? opProcessorStopReason(stopProcessor)
262  : OP_SR_EXIT;
263  switch(sr) {
264 
265  case OP_SR_EXIT:
266  case OP_SR_FINISH:
267  finished = True;
268  break;
269 
270  case OP_SR_BP_ICOUNT:
271  opPrintf(
272  "Processor %s icount %u stopped at icount\n",
273  opObjectName (stopProcessor),
274  (Uns32)opProcessorICount(stopProcessor)
275  );
276  break;
277 
278  case OP_SR_BP_ADDRESS:
279  opPrintf(
280  "Processor %s icount %u breakpoint at address 0x%08x\n",
281  opObjectName (stopProcessor),
282  (Uns32)opProcessorICount(stopProcessor),
283  (Uns32)opProcessorPC(stopProcessor)
284  );
285  stepOver = True;
286  break;
287 
288  case OP_SR_WATCHPOINT:
289  opPrintf(
290  "Processor %s icount %u stopped at watchpoint\n",
291  opObjectName (stopProcessor),
292  (Uns32)opProcessorICount(stopProcessor)
293  );
294  handleWatchpoints(root);
295  break;
296 
297  default:
298  opPrintf(
299  "Processor %s icount %u stopped with unexpected reason 0x%x (%s)\n",
300  opObjectName (stopProcessor),
301  (Uns32)opProcessorICount(stopProcessor),
302  sr,
303  opStopReasonString(sr)
304  );
305  break;
306  }
307  }
308 
309  opSessionTerminate();
310 
311 
312  return (opErrors() ? 1 : 0); // set exit based upon any errors
313 }
