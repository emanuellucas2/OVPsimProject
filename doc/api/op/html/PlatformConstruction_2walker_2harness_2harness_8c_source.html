53 int main(int argc, const char *argv[]) {
54  opSessionInit(OP_VERSION);

56  optCmdParserP parser = opCmdParserNew(HARNESS_NAME, OP_AC_ALL);
57  opCmdParserAdd(parser, "modulepath" , "", "file", "input", OP_FT_STRINGVAL, &options.modulepath, "path to a module", OP_AC_ALL, 0, 0);
58  opCmdParserAdd(parser, "objectname" , "", "path", "input", OP_FT_STRINGVAL, &options.hierName, "hierarchical object name", OP_AC_ALL, 0, 0);
59  opCmdParseArgs(parser, argc, argv);
60 
61  if(opErrors() == 0) {
62 
63  optModuleP mi = opRootModuleNew(0, 0, 0);
64  if(!options.modulepath) {
65  options.modulepath = MODULE_DIR;
66  }
67  opModuleNew(mi, options.modulepath, MODULE_INSTANCE, 0, 0);
68 
69  opRootModulePreSimulate(mi);
70  opRootModuleTimeAdvance(mi, 0);
71 
72  // call the walker
73  if(options.hierName) {
74  optObjectP startObject = opObjectByName(mi, options.hierName, 0);
75 
76  // Was an object found - need to check [any] member of the union for non-zero?
77  if(startObject.Module) {
78  walkObject(0, startObject);
79  } else {
80  opMessage("E", "HARNESS_ONF", "Object '%s' not found", options.hierName);
81  }
82  } else {
83  walkModule(0, mi);
84  }
85  }
86 
87  opSessionTerminate();

212 static void walkFormal(Uns32 indent, optFormalP o) {
213  indentText(indent++);
214  opPrintf(
215  "Formal: \"%s\" (%s)\n",
216  opObjectHierName(o),
217  opFormalSystem(o) ? "system" : "model"
218  );
219 
220  indentText(indent);
221  opPrintf("Type: \"%s\"\n", opFormalTypeString(o));
222  indentText(indent);
223  opPrintf("Desc: \"%s\"\n", opFormalDescription(o));
224 
225  if(opFormalType(o) == OP_PARAM_ENUM) {
226  walkFormalEnums(indent, o);
227  } else {
228 
229  switch(opFormalType(o)) {
230  case OP_PARAM_BOOL :
231  indentText(indent);
232  opPrintf("default:%u\n", opFormalBoolDefaultValue(o));
233  break;
234 
235  case OP_PARAM_INT32: {
236  indentText(indent);
237  Int32 min,max,dflt;
238  opFormalInt32Limits(o, &min, &max, &dflt);
239  if(min == max) {
240  opPrintf("[no limits] default:%d\n", dflt);
241  } else {
242  opPrintf("min:%d max:%d default:%d\n", min, max, dflt);
243  }
244  break;
245  }
246  case OP_PARAM_INT64: {
247  indentText(indent);
248  Int64 min,max,dflt;
249  opFormalInt64Limits(o, &min, &max, &dflt);
250  if(min == max) {
251  opPrintf("[no limits] default:" FMT_64d "\n", dflt);
252  } else {
253  opPrintf("min:" FMT_64d " max:" FMT_64d " default:" FMT_64d "\n", min, max, dflt);
254  }
255  break;
256  }
257  case OP_PARAM_UNS32: {
258  indentText(indent);
259  Uns32 min,max,dflt;
260  opFormalUns32Limits(o, &min, &max, &dflt);
261  if(min == max) {
262  opPrintf("[no limits] default:%u\n", dflt);
263  } else {
264  opPrintf("min:%u max:%u default:%u\n", min, max, dflt);
265  }
266  break;
267  }
268 
269  case OP_PARAM_UNS64: {
270  indentText(indent);
271  Uns64 min,max,dflt;
272  opFormalUns64Limits(o, &min, &max, &dflt);
273  if(min == max) {
274  opPrintf("[no limits] default:" FMT_64u "\n", dflt);
275  } else {
276  opPrintf("min:" FMT_64u " max:" FMT_64u " default:" FMT_64u "\n", min, max, dflt);
277  }
278  break;
279  }
280 
281  case OP_PARAM_DOUBLE: {
282  indentText(indent);
283  double min,max,dflt;
284  opFormaldoubleLimits(o, &min, &max, &dflt);
285  if(min == max) {
286  opPrintf("[no limits] default:" FMT_DBL "\n", dflt);
287  } else {
288  opPrintf("min:" FMT_DBL " max:" FMT_DBL " default:" FMT_DBL "\n", min, max, dflt);
289  }
290  break;
291  }
292 
293  case OP_PARAM_STRING :
294  indentText(indent);
295  opPrintf("max:%u default:(%s)\n", opFormalStringMaxLength(o), opFormalStringDefaultValue(o));
296  break;
297 
298  default :
299  break;
300  }
301  }
302 }

304 static void walkParam(Uns32 indent, optParamP o) {
305  indentText(indent++);
306  if(opParamType(o) == OP_PARAM_NONE) {
307  opPrintf("Parameter: (cursor)\n");
308  } else {
309 
310  const char *name = opObjectName(o);
311  optObjectP parent = opObjectParent(o);
312  Bool set = False;
313 
314  opPrintf("Parameter: \"%s\" ", opObjectHierName(o));
315 
316  switch(opParamType(o)) {
317  case OP_PARAM_BOOL : opPrintf("%u (%s)\n", opObjectParamBoolValue (parent, name, &set), opObjectParamStringValue(parent, name)); break;
318  case OP_PARAM_INT32 : opPrintf("%d (%s)\n", opObjectParamInt32Value (parent, name, &set), opObjectParamStringValue(parent, name)); break;
319  case OP_PARAM_INT64 : opPrintf(FMT_64d " (%s)\n", opObjectParamInt64Value (parent, name, &set), opObjectParamStringValue(parent, name)); break;
320  case OP_PARAM_UNS32 : opPrintf("%u (%s)\n", opObjectParamUns32Value (parent, name, &set), opObjectParamStringValue(parent, name)); break;
321  case OP_PARAM_UNS64 : opPrintf(FMT_64u " (%s)\n", opObjectParamUns64Value (parent, name, &set), opObjectParamStringValue(parent, name)); break;
322  case OP_PARAM_DOUBLE : opPrintf(FMT_DBL " (%s)\n", opObjectParamDoubleValue(parent, name, &set), opObjectParamStringValue(parent, name)); break;
323  case OP_PARAM_ENDIAN : opPrintf("%u (%s)\n", opObjectParamEndianValue(parent, name, &set), opObjectParamStringValue(parent, name)); break;
324  case OP_PARAM_ENUM : opPrintf("%u (%s)\n", opObjectParamEnumValue (parent, name, &set), opObjectParamStringValue(parent, name)); break;
325  case OP_PARAM_PTR : opPrintf("0x%x (%s)\n", (Uns32)(UnsPS)opObjectParamPtrValue(parent, name, &set), opObjectParamStringValue(parent, name)); break;
326  case OP_PARAM_STRING : opPrintf("%s\n", opObjectParamStringValue(parent, name)); break;
327  default : opPrintf("(unable to print)\n");
328  }
329  }
