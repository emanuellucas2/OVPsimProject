39 static void cmdParser(optCmdParserP parser) {
40  opCmdParserAdd(parser, "myrecord", "", "bool", "myargs group", OP_FT_BOOLVAL,
41  &options.record, "Record from UART to file", OP_AC_ALL, 0, 0);
42  opCmdParserAdd(parser, "myreplay", "", "bool", "myargs group" , OP_FT_BOOLVAL,
43  &options.replay, "Replay UART from file", OP_AC_ALL, 0, 0);
44  opCmdParserAdd(parser, "myuseport", "", "bool", "myargs group" , OP_FT_BOOLVAL,
45  &options.useport, "Use manual port connection", OP_AC_ALL, 0, 0);
46 }

48 int main(int argc, const char *argv[]) {
49  opSessionInit(OP_VERSION);
50 
51  optCmdParserP parser = opCmdParserNew(HARNESS_NAME, OP_AC_ALL);
52  cmdParser(parser);
53  opCmdParseArgs(parser, argc, argv);
54 
55  opParamBoolOverride(0, OP_FP_STOPONCONTROLC, 1);
56  if (options.record) {
57  opParamStringOverride(0, MODULE_INSTANCE "/" PERIPHERAL_INSTANCE "/" "record", "recdir/uart0.record");
58  } else if (options.replay) {
59  opParamStringOverride(0, MODULE_INSTANCE "/" PERIPHERAL_INSTANCE "/" "replay", "recdir/uart0.record");
60  }
61  if (options.useport) {
62  opParamUns32Override(0, MODULE_INSTANCE "/" PERIPHERAL_INSTANCE "/" "portnum", 0);
63  } else {
64  opParamBoolOverride (0, MODULE_INSTANCE "/" PERIPHERAL_INSTANCE "/" "console", 1);
65  }
66  // stop when a UART console terminal is closed
67  opParamBoolOverride (0, MODULE_INSTANCE "/" PERIPHERAL_INSTANCE "/" "finishOnDisconnect", 1);
68 
69  optModuleP mi = opRootModuleNew(0, 0, 0);
70  opModuleNew(mi, MODULE_DIR, MODULE_INSTANCE, 0, 0);
71 
72  opRootModuleSimulate(mi);
73 
74  opSessionTerminate();
75 
76  return (opErrors() ? 1 : 0); // set exit based upon any errors
77 }
