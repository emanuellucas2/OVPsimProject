35 int main(int argc, const char *argv[]) {
36  opSessionInit(OP_VERSION);
37 
38  opCmdParseStd (argv[0], OP_AC_ALL, argc, argv);
39 
40  optModuleP mr = opRootModuleNew(0, 0, 0);
41  optModuleP mi = opModuleNew(mr, "module", "u1", 0, 0);
42 
43  opRootModulePreSimulate(mr);
44 
45  // get processor, assumes only a single processor
46  optProcessorP processor = opProcessorNext(mi, NULL);
47 
48  // Run the processor for 100 instructions
49  opProcessorSimulate(processor, 100);
50 
51  // Call a command that had been created in a processor model
52  // Pointer to the call command result string. This could be any string
53  const char *result;
54 
55  opPrintf("\nCall Model Commands\n\n");
56 
57  // This shows how to call a command using its full name, starting at the root
58  // (note that the platfrom root 'mr' is passed).
59  // opCommandStringCall() returns NULL if the command failed or the string returned by the command.
60 
61  // set up the arguments
62  const char *cmd0Argv[] = {"mipsDebugFlags", "-help"};
63  Uns32 cmd0Argc = sizeof(cmd0Argv)/sizeof(const char *);
64  result = opCommandStringCall(mr, "u1/cpu1/mipsDebugFlags", cmd0Argc, cmd0Argv);
65  opPrintf("\nopCommandStringCall result '%s'\n\n", result);
66 
67  // Setup command for reading MIPS COP0 registers
68  const char* cmdToExecute = "u1/cpu1/mipsCOP0";
69  optCommandP cmd = opObjectByName(mr, cmdToExecute, OP_COMMAND_EN).Command;
70  if(!cmd)
71  opMessage("F", "harness", "Command '%s' not found", cmdToExecute);
72 
73  // Setup calling arguments to read COP0 'Config' register
74  const char *cmd1Argv[] = {"mipsCOP0", "-register", "16", "-select", "0"};
75  Uns32 cmd1Argc = sizeof(cmd1Argv)/sizeof(const char *);
76 
77  result = opCommandCall( cmd, cmd1Argc, cmd1Argv);
78 
79  opPrintf("\nCall Command result '%s'\n", result);
80 
81  // Setup calling arguments to read COP0 'PrId' register
82  const char *cmd2Argv[] = {"mipsCOP0", "-register", "15", "-select", "0"};
83  Uns32 cmd2Argc = sizeof(cmd2Argv)/sizeof(const char *);
84 
85  result = opCommandCall( cmd, cmd2Argc, cmd2Argv);
86 
87  opPrintf("\nCall Command result '%s'\n", result);
88 
89  opPrintf("\nDiscover installed commands\n");
90 
91  // iterates over all commands on all processor instances
92  opModuleCommandsShow(mi);
93 
94  const char *procName = "cpu1";
95  opPrintf("Commands for %s\n", procName);
96  optProcessorP proc = opObjectByName(mi, procName, OP_PROCESSOR_EN).Processor;
97 
98  // iterates over all commands on this processor instance
99  opProcessorCommandIterAll(proc, printCommand, 0);
100 
101  opPrintf("\nComplete Simulation\n\n");
102 
103  // run simulation to completion
104  opRootModuleSimulate(mr);
105 
106  // In this case the wrong root is passed to opCommandStringCall.
107  // The correct
108  opPrintf("\nNow trying to call a command giving the wrong root:\n\n");
109  result = opCommandStringCall(mi, "u1/cpu1/mipsDebugFlags", cmd0Argc, cmd0Argv);
110  opPrintf("\nopCommandStringCall returned '%s'\n\n", result);
111 
112  opSessionTerminate();
113 
114  return (opErrors() ? 1 : 0); // set exit based upon any errors
115 }
