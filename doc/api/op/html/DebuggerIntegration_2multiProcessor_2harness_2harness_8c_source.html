93 //
94 // Apply a watchpoint on a processor memory location
95 //
96 static void applyWatchpoint(
97  optProcessorP proc,
98  Addr address,
99  Addr size,
100  watchType type
101 ) {
102 
103  if (type == WATCH_RW) {
104 
105  // create access watchpoint
106  optWatchpointP rwp = opProcessorAccessWatchpointNew(
107  proc, False, address, address+size-1, (void *)(id++), 0
108  );
109  opPrintf("%s: ACCESS watchpoint is %u\n", opObjectName(proc), getWatchpointId(rwp));
110 
111  } else if (type == WATCH_READ) {
112 
113  // create read watchpoint
114  optWatchpointP rwp = opProcessorReadWatchpointNew(
115  proc, False, address, address+size-1, (void *)(id++), 0
116  );
117  opPrintf("%s: READ watchpoint is %u\n", opObjectName(proc), getWatchpointId(rwp));
118 
119  } else if (type == WATCH_WRITE) {
120 
121  // create write watchpoint
122  optWatchpointP wwp = opProcessorWriteWatchpointNew(
123  proc, False, address, address+size-1, (void *)(id++), 0
124  );
125  opPrintf("%s: WRITE watchpoint is %u\n", opObjectName(proc), getWatchpointId(wwp));
126 
127  }
128 }

154 //
155 // Query registers and register groups in the passed processor
156 //
157 static void queryRegisters(optProcessorP processor) {
158 
159  opPrintf("%s REGISTERS\n", opObjectName(processor));
160 
161  optRegGroupP group = 0;
162 
163  while((group=opProcessorRegGroupNext(processor, group))) {
164 
165  opPrintf(" GROUP %s\n", opRegGroupName(group));
166 
167  optRegP reg = 0;
168 
169  while((reg=opRegGroupRegNext(processor, group, reg))) {
170  opPrintf(" REGISTER %s\n", opRegName(reg));
171  }
172  }
173 }

175 //
176 // Handle all active watchpoints
177 //
178 static void handleWatchpoints(optModuleP mi) {
179 
180  optWatchpointP wp;
181 
182  while((wp=opRootModuleWatchpointNext(mi))) {
183 
184  Uns32 id = getWatchpointId(wp);
185  optProcessorP processor = opWatchpointTriggeredBy(wp);
186  optWatchpointType type = opWatchpointType(wp);
187 
188  switch(type) {
189 
190  case OP_WP_REG: {
191 
192  // a register watchpoint was triggered
193  optRegP reg = opWatchpointReg(wp);
194  Uns32 *newValueP = (Uns32 *)opWatchpointRegCurrentValue(wp);
195  Uns32 *oldValueP = (Uns32 *)opWatchpointRegPreviousValue(wp);
196 
197  // indicate old and new value of the affected register
198  opPrintf(
199  " Register watchpoint %u (processor %s:%s) triggered 0x%08x->0x%08x\n",
200  id,
201  opObjectName(processor),
202  opRegName(reg),
203  *oldValueP,
204  *newValueP
205  );
206 
207  // delete watchpoint after 100 triggerings
208  if(regWatchPointCount++>100) {
209  opWatchpointDelete(wp);
210  } else {
211  opWatchpointReset(wp);
212  }
213 
214  break;
215  }
216 
217  case OP_WP_MEM_READ:
218  case OP_WP_MEM_WRITE:
219  case OP_WP_MEM_ACCESS: {
220 
221  // a memory watchpoint was triggered
222  opPrintf(
223  " %s watchpoint %u (range 0x%08x:0x%08x) triggered by processor %s\n",
224  (type==OP_WP_MEM_READ) ? "Read" : (type==OP_WP_MEM_WRITE) ? "Write" : "Access",
225  id,
226  (Uns32)opWatchpointAddressLo(wp),
227  (Uns32)opWatchpointAddressHi(wp),
228  opObjectName(processor)
229  );
230 
231  opWatchpointReset(wp);
232 
233  break;
234  }
235 
236  default: {
237 
238  opPrintf(
239  " unknown watchpoint type %u triggered by processor %s\n",
240  type,
241  opObjectName (processor)
242  );
243 
244  opWatchpointReset(wp);
245 
246  break;
247  }
248  }
249  }
250 }

252 //
253 // Main routine
254 //
255 int main(int argc, const char *argv[]) {
256 
257  // Start up simulator and parse command line arguments
258  opSessionInit(OP_VERSION);
259  opCmdParseStd (argv[0], OP_AC_ALL, argc, argv);
260 
261  // Load the design
262  optModuleP root = opRootModuleNew(0, 0, 0);
263  opModuleNew(root, MODULE_DIR, MODULE_INSTANCE, 0, 0);
264 
265  // Get processor handles
266  optProcessorP processor0 = opObjectByName(root, MODULE_INSTANCE "/" CPU0_INSTANCE, OP_PROCESSOR_EN).Processor;
267  optProcessorP processor1 = opObjectByName(root, MODULE_INSTANCE "/" CPU1_INSTANCE, OP_PROCESSOR_EN).Processor;
268 
269  // Load application programs
270  opProcessorApplicationLoad(processor0, APP_DIR "/" APP_ENCRYPT, APP_LDR_CTRL, 0);
271  opProcessorApplicationLoad(processor1, APP_DIR "/" APP_DECRYPT, APP_LDR_CTRL, 0);
272 
273  // Must advance simulator phase before setting watchpoints
274  opRootModulePreSimulate(root);
275 
276  // Apply watchpoint on writes to the first word of shared memory address (ENCRYPT_INDEX)
277  // (only needs to be added for one of the processors since it is shared memory)
278  applyWatchpoint(processor0, SHARED_LOW, 4, WATCH_WRITE);
279 
280  // Set register watchpoints for processor0
281  applyRegWatchpoints(processor0);
282 
283  // Report registers and register groups in processor0
284  queryRegisters(processor0);
285 
286  Bool stepOver = False;
287  Bool finished = False;
288  optProcessorP stopProcessor = 0;
289 
290  // Simulation loop
291  while(!finished) {
292 
293  // Simulate the platform using the default scheduler
294  // All breakpoints are set and cleared together, ala gdb behavior
295  if(stepOver) {
296  opProcessorBreakpointICountSet(stopProcessor, 1);
297  stopProcessor = opRootModuleSimulate (root);
298  stepOver = False;
299  } else {
300  applyBreakpoints(processor0, breakpoints0);
301  applyBreakpoints(processor1, breakpoints1);
302  stopProcessor = opRootModuleSimulate(root);
303  clearBreakpoints(processor0, breakpoints0);
304  clearBreakpoints(processor1, breakpoints1);
305  }
306 
307  optStopReason sr = stopProcessor ? opProcessorStopReason(stopProcessor)
308  : OP_SR_EXIT;
309  switch(sr) {
310 
311  case OP_SR_EXIT:
312  case OP_SR_FINISH:
313  finished = True;
314  break;
315 
316  case OP_SR_BP_ICOUNT:
317  opPrintf(
318  "Processor %s icount %u stopped at icount\n",
319  opObjectName (stopProcessor),
320  (Uns32)opProcessorICount(stopProcessor)
321  );
322  break;
323 
324  case OP_SR_BP_ADDRESS:
325  opPrintf(
326  "Processor %s icount %u breakpoint at address 0x%08x\n",
327  opObjectName (stopProcessor),
328  (Uns32)opProcessorICount(stopProcessor),
329  (Uns32)opProcessorPC(stopProcessor)
330  );
331  stepOver = True;
332  break;
333 
334  case OP_SR_WATCHPOINT:
335  opPrintf(
336  "Processor %s icount %u stopped at watchpoint\n",
337  opObjectName (stopProcessor),
338  (Uns32)opProcessorICount(stopProcessor)
339  );
340  handleWatchpoints(root);
341  break;
342 
343  default:
344  opPrintf(
345  "Processor %s icount %u stopped with unexpected reason 0x%x (%s)\n",
346  opObjectName (stopProcessor),
347  (Uns32)opProcessorICount(stopProcessor),
348  sr,
349  opStopReasonString(sr)
350  );
351  break;
352  }
353  }
354 
355  opSessionTerminate();
356 
357 
358  return (opErrors() ? 1 : 0); // set exit based upon any errors
359 }
