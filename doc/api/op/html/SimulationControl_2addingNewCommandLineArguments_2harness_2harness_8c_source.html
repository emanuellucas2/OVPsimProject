41 static void cmdParser(optCmdParserP parser) {
42  opCmdParserAdd(parser, "myarg", "m", "bool", "myargs group", OP_FT_BOOLVAL, &options.myarg, "This is my description [default: 0]", OP_AC_ALL, 0, 0);
43  opCmdParserAdd(parser, "newstring", 0 , "string", "myargs group", OP_FT_STRINGVAL, &options.newstring, "A new string", OP_AC_ALL, 0, 0);

47 static void cmdParserCheck(optCmdParserP parser) {
48  if (!opCmdArgUsed (parser,"program")) {
49  opMessage("F", HARNESS_NAME, "Argument '--program' must be specified");
50  }

53  }
54  if (options.newstring) {
55  opMessage("I", HARNESS_NAME, "Argument 'newstring' specified (\"%s\")", options.newstring);
56  }
57 }
