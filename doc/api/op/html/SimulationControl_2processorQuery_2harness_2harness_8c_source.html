19 /*
20  * This example shows how information can be extracted from a processor model.
21  * This information can be used to
22  * : create documentation
23  * : build a platform that makes few assumptions about the model.
24  * : build a debugger that adapts to the model.
25  *
26  * Note that each processor model can return different information, as can
27  * different variants of the same model.
28  *
29  * Note that some functions shown here that return human readable strings have
30  * sister functions that return enumerated types. You should not have to compare
31  * strings to identify the types of registers, ports, modes etc.
32  */

90 //
91 // Callback for register query. This gets information about each register and how
92 // it can be used.
93 //
94 static OP_PROCESSOR_ITERATOR_FN(queryRegistersCB) {
95 
96  opPrintf(" REGISTERS\n");
97 
98  optRegGroupP group = NULL;
99 
100  // Are the registers in groups
101  if(!opProcessorRegGroupNext(processor, group)) {
102 
103  // no register groups
104  optRegP reg = NULL;
105 
106  if(!opProcessorRegNext(processor, reg)) {
107 
108  opPrintf(" NO REGISTER INFORMATION\n");
109 
110  } else {
111 
112  while((reg=opProcessorRegNext(processor, reg))) {
113 
114  const char *name = opRegName(reg);
115  const char *desc = opRegDescription(reg);
116  Uns32 regbits = opRegBits(reg);
117 
118  // Usage as a string, (see opRegUsageEnum)
119  const char *use = opRegUsageString(reg);
120 
121  const char *readonly = opRegReadOnly(reg) ? "readonly" : "readwrite";
122 
123  opPrintf(" REGISTER (%2d bits) %-15s %-25s %-10s", regbits, use, name, readonly);
124  if (desc) {
125  opPrintf(" (%s)", desc);
126  }
127  opPrintf("\n");
128  }
129  }
130 
131  } else {
132  // iterate over register groups
133  while((group=opProcessorRegGroupNext(processor, group))) {
134 
135  opPrintf(" GROUP %s\n", opRegGroupName(group));
136 
137  optRegP reg = NULL;
138 
139  if(!opRegGroupRegNext(processor, group, reg)) {
140 
141  opPrintf(" NO REGISTER INFORMATION\n");
142 
143  } else {
144 
145  while((reg=opRegGroupRegNext(processor, group, reg))) {
146 
147  const char *name = opRegName(reg);
148  const char *desc = opRegDescription(reg);
149  Uns32 regbits = opRegBits(reg);
150 
151  // Usage as a string, (see opRegUsageEnum)
152  const char *use = opRegUsageString(reg);
153 
154  const char *readonly = opRegReadOnly(reg) ? "readonly" : "readwrite";
155  opPrintf(" REGISTER (%2d bits) %-15s %-25s %-10s", regbits, use, name, readonly);
156  if (desc) {
157  opPrintf(" (%s)", desc);
158  }
159  opPrintf("\n");
160  }
161  }
162  }
163  }
164  opPrintf("\n");
165 }

352 //
353 // Print the description and name of this processor, then recursively find all children
354 // Note the the description field is optional and is generally missing from single-core processors.
355 //
356 static void queryDescriptionCB(optProcessorP processor, Uns32 depth) {
357 
358  const char *desc = opProcessorDescription(processor);
359  const char *name = opObjectName(processor);
360 
361  indentText(depth++);
362  opPrintf(" %s:%s\n", name, desc ?: "(no description)");
363 
364  optProcessorP ch;
365  for (ch= opProcessorChild(processor); ch ; ch = opProcessorSiblingNext(ch)) {
366  queryDescriptionCB(ch, depth);
367  }
368 }

387 //
388 // Get other information embedded in the model
389 //
390 static void queryInfo(optProcessorP processor) {
391 
392  // the path to the gdb in the Imperas installation
393  // This will select the correct gdb regardless of processor variant, host OS etc..,.
394  const char *gdbpath = opProcessorGdbPath(processor);
395 
396  // Any special flags required by the gdb when used in this context.
397  const char *gdbflags = opProcessorGdbFlags(processor);
398 
399  // This string is usually vendor-specific.
400  // Other processors in this family will have the same string.
401  const char *family = opProcessorFamily(processor);
402 
403  // Where the processor is expecting to be stored. This can usually be ignored
404  // since you knew where to find the processor in the first place.
405  optVLNVP model = opObjectVLNV(processor);
406 
407  // A semihost library to be used with this processor by default.
408  optVLNVP semihost = opProcessorDefaultSemihost(processor);
409 
410  // Which endian will this processor support? This is NOT the current endian.
411  optEndian endian = opProcessorEndian(processor, True);
412 
413  // A list of ELF codes that this processor supports. Used in the case where
414  // one or more GNU (or compatible) tool-chains are used to generate code for
415  // this processor.
416  Bool defaultElf = 0;
417  Uns32 max = 10;
418  Uns32 elves[max];
419  Uns32 actualElves = opProcessorElfCodes(processor, elves, max, &defaultElf);
420  if(actualElves > max) {
421  // truncate the result - could allocate a larger array if required.
422  actualElves = max;
423  }
424 
425  opPrintf(" GENERAL INFORMATION\n");
426  opPrintf(" GDB path : %s\n", gdbpath ?: "(none)");
427  opPrintf(" GDB flags : %s\n", gdbflags ?: "(none)");
428  opPrintf(" Processor family : %s\n", family ?: "(none)");
429  opPrintf(" Processor endian : %d\n", endian);
430 
431  printVLNV(" Model VLNV :", model);
432  printVLNV(" Semihost VLNV :", semihost);
433 
434  opPrintf(" ELF Code(s) :");
435  Uns32 i;
436  for(i=0; i < actualElves; i++) {
437  opPrintf(" %d", elves[i]);
438  }
439  if(defaultElf) {
440  opPrintf(" (This is the default processor for these ELF codes)\n");
441  }
442  opPrintf("\n");
443 }

445 //
446 // A model can supply documentation with arbitrary depth and detail. It is up the
447 // the consumer to format it as desired. The data is presented in a structure similar
448 // to the DOM used by some XML documents.
449 //
450 static void queryDocumentationNode(optDocNodeP node, Uns32 depth) {
451  optDocNodeType type = opDocNodeType(node);
452  indentText(depth++);
453  switch(type) {
454  case OP_DOC_NODE_TITLE :
455  opPrintf("%s\n", opDocText(node));
456  break;
457  case OP_DOC_NODE_TEXT :
458  opPrintf("%s\n", opDocText(node));
459  break;
460  case OP_DOC_NODE_FIELDS:
461  opPrintf("%s w:%u\n", opDocText(node), opDocFieldWidth(node));
462  break;
463  case OP_DOC_NODE_FIELD :
464  opPrintf("%s w:%u o:%u\n", opDocText(node), opDocFieldWidth(node), opDocFieldOffset(node));
465  break;
466  }
467  optDocNodeP ch = 0;
468  while((ch = opDocChildNext(node, ch))) {
469  queryDocumentationNode(ch, depth);
470  }
471 }

507 static void queryProcessor(
508  const char *instance,
509  const char *vendor,
510  const char *name,
511  const char *endian,
512  const char *variant
513 ) {
514 
515  opSessionInit(OP_VERSION);
516 
517  optModuleP mr = opRootModuleNew(0, HARNESS_NAME, 0);
518 
519  const char *path = opVLNVString(NULL, vendor, "processor", name, "1.0", OP_PROCESSOR, OP_VLNV_FATAL);
520 
521  if(path) {
522 
523  optProcessorP processor = newProcessor(mr, instance, path, endian, variant);
524 
525  if(processor) {
526  opPrintf("PROCESSOR: Vendor: %s Name: %s Variant: %s\n", vendor, name, variant?: "(no variant)");

528  queryNets(processor);
529  queryRegisters(processor);
530  queryExceptions(processor);
531  queryModes(processor);
532  queryFormals(processor);
533  queryDescription(processor);
534  queryInfo(processor);
535  queryDocumentation(processor);
536  }
537  }
538 
539  opSessionTerminate();
