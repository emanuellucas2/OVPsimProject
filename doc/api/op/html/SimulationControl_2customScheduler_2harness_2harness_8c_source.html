34 // Define processor performance and time information
35 // Processor MIPS rate 10 MIPS (see hardware module defintion)
36 // The default internal scheduler uses
37 // Quantum 1 mS
38 // But can be configured using the OP_FP_QUANTUM parameter
39 // Configure in this example so we can see the UART access in different time slices, by using
40 // Quantum 0.01 mS
41 
42 #define INSTRUCTIONS_PER_SECOND 10000000
43 #define QUANTUM_TIME_SLICE 0.00001
44 #define INSTRUCTIONS_PER_TIME_SLICE (INSTRUCTIONS_PER_SECOND*QUANTUM_TIME_SLICE)
45 
46 int main(int argc, const char *argv[]) {
47 
48  opSessionInit(OP_VERSION);
49 
50  optCmdParserP parser = opCmdParserNew(HARNESS_NAME, OP_AC_ALL);
51  opCmdParserAdd(parser, "custom", 0, 0, "user", OP_FT_BOOLVAL,
52  &options.custom,
53  "Use a custom scheduler instead of built-in scheduler",
54  OP_AC_ALL, 0, 0);
55  if (!opCmdParseArgs(parser, argc, argv)) {
56  opMessage("E", HARNESS_NAME, "Command line parse incomplete");
57  }
58 
59  // create the root module with reduced Quantum (in line with Custom Scheduler)
60  optParamP params = OP_PARAMS (OP_PARAM_DOUBLE_SET(OP_FP_QUANTUM, QUANTUM_TIME_SLICE));
61  optModuleP mi = opRootModuleNew(0, 0, params);
62  optModuleP ui = opModuleNew(mi, MODULE_DIR, MODULE_INSTANCE, 0, 0);
63 
64  // get first processor (assumes only one processor in module)
65  optProcessorP proc = opProcessorNext(ui, NULL);
66 
67  opMessage(
68  "I", HARNESS_NAME,
69  "Running with %s scheduler",
70  options.custom ? "custom" : "standard"
71  );
72 
73  // must advance to next phase for the API calls that follow
74  opRootModulePreSimulate(mi);
75 
76  if (options.custom) {
77 
78  // run simulation with custom scheduling
79  optTime myTime = QUANTUM_TIME_SLICE;
80  optStopReason stopReason = OP_SR_SCHED;
81  do {
82 
83  // move time forward by time slice on root module
84  // NOTE: This matches the standard scheduler which moves time forward in
85  // the system and then executes instructions on all processors
86  opRootModuleTimeAdvance(mi, myTime);
87  opMessage(
88  "I", HARNESS_NAME,
89  "Advance Time to %g seconds",
90  (double)myTime
91  );
92 
93  // run processor for number of instructions calculated for time slice
94  stopReason = opProcessorSimulate(proc, INSTRUCTIONS_PER_TIME_SLICE);
95  if ((stopReason!=OP_SR_SCHED) && (stopReason!=OP_SR_HALT)) {
96 
97  opMessage(
98  "I", HARNESS_NAME,
99  "Simulation Complete (%s)",
100  opStopReasonString(stopReason)
101  );
102 
103  break; // finish simulation loop
104 
105  }
106 
107  myTime += QUANTUM_TIME_SLICE;
108 
109  } while (1);
110 
111  } else {
112 
113  // run with built in standard scheduler
114  opRootModuleSimulate(mi);
115 
116  }
117 
118  opMessage(
119  "I", HARNESS_NAME,
120  "Time at Completion %g seconds",
121  (double)opModuleCurrentTime(mi)
122  );
123 
124  opSessionTerminate();
125 
126  return (opErrors() ? 1 : 0); // set exit based upon any errors
127 
128 }
