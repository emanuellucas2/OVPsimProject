33 // Give the monitoring a threshold
34 static unsigned int count;
35 #define MONITOR_THRESHOLD 100
36 
37 //
38 // triggered when registered access happens and prints information of access
39 //
40 static OP_MONITOR_FN(monitorCallback) {
41  if(++count < MONITOR_THRESHOLD) {
42  opMessage ("I", PREFIX "_MT", "Monitor triggered (%03d): "
43  "callback '%s': processor '%s' : "
44  "type '%s' : bytes %u : "
45  "address Physical 0x" FMT_A0Nx" Virtual 0x" FMT_A0Nx " %s",
46  count,
47  __FUNCTION__,
48  processor ? opObjectName(processor) : "artifact", // if no processor this is an artifact access
49  (const char*)userData,
50  bytes,
51  addr,
52  VA,
53  (count == MONITOR_THRESHOLD) ? "(threshold reached)" : "");
54  }
55 }

57 //
58 // iterate across the buses found in the module and register callbacks for read, write and fetch
59 //
60 static void monitorBus(optModuleP mi) {
61  // iterate across all busses found in module
62  optBusP bus = 0;
63  while ((bus = opBusNext(mi, bus))) {
64  Addr min = 0;
65  Addr max = opBusMaxAddress(bus);
66  opMessage("I", PREFIX "_BM", "Add monitor for '%s' (0x" FMT_A0Nx " to 0x" FMT_A0Nx ")\n",
67  opObjectHierName(bus), (Addr)0, max);
68  opBusFetchMonitorAdd(bus, 0, min, max, monitorCallback, "bus-fetch");
69  opBusReadMonitorAdd (bus, 0, min, max, monitorCallback, "bus-read");
70  opBusWriteMonitorAdd(bus, 0, min, max, monitorCallback, "bus-write");
71  }
72 }
