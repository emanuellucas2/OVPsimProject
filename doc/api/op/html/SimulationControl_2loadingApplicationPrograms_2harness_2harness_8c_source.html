31 struct optionsS {
32  Bool analyzeinstances;
33  // harness application load options
34  Bool loadprocessor;
35  Bool loadmemory;
36  Bool loadbus;
37  const char * application;
38 } options = {
39  .analyzeinstances = False,
40  .loadprocessor = False,
41  .loadmemory = False,
42  .loadbus = False,
43  .application = NULL,
44 };

46 int main(int argc, const char *argv[]) {
47  opSessionInit(OP_VERSION);
48 
49  optCmdParserP parser = opCmdParserNew(HARNESS_NAME, OP_AC_ALL);
50  opCmdParserAdd(parser, "loadprocessor", 0, 0, "user", OP_FT_BOOLVAL, &options.loadprocessor,
51  "load the application elf file onto a processor instance.", OP_AC_ALL, 0, 0);
52  opCmdParserAdd(parser, "loadmemory", 0, 0, "user", OP_FT_BOOLVAL, &options.loadmemory,
53  "load the application elf file onto a memory instance.", OP_AC_ALL, 0, 0);
54  opCmdParserAdd(parser, "loadbus", 0, 0, "user", OP_FT_BOOLVAL, &options.loadbus,
55  "load the application elf file onto a bus instance.", OP_AC_ALL, 0, 0);
56  opCmdParserAdd(parser, "application", 0, "string", "user", OP_FT_STRINGVAL, &options.application,
57  "the application elf file to load.", OP_AC_ALL, 0, 0);
58  opCmdParserAdd(parser, "analyzeinstances", 0, "bool", "user", OP_FT_BOOLVAL, &options.analyzeinstances,
59  "Report analysis of applications loaded onto processor.", OP_AC_ALL, 0, 0);
60 
61  if (!opCmdParseArgs(parser, argc, argv)) {
62  opMessage("E", HARNESS_NAME, "Command line parse incomplete");
63  }
64 
65  optModuleP mi = opRootModuleNew(0, 0, 0);
66 
67  optModuleP ui = opModuleNew(mi, MODULE_DIR, MODULE_INSTANCE, 0, 0);
68 
69  // Get first Processor (assumes a single processor module)
70  optProcessorP proc = opProcessorNext(ui, NULL);
71  optApplicationP thisApp = NULL;
72 
73  // Loading application
74  if (options.loadprocessor ) {
75 
76  // load on processor, setting start address
77  opProcessorApplicationLoad(
78  proc, options.application, OP_LDR_VERBOSE|OP_LDR_FAIL_IS_ERROR|OP_LDR_SET_START, 0
79  );
80 
81  } else if (options.loadmemory) {
82 
83  // load on first Memory (assumes a single memory module)
84  optMemoryP mem = opMemoryNext(ui, NULL);
85  thisApp = opMemoryApplicationLoad(
86  mem, options.application, OP_LDR_VERBOSE|OP_LDR_FAIL_IS_ERROR, 0
87  );
88 
89  } else if (options.loadbus) {
90 
91  // load on first Bus (assumes a single bus module)
92  optBusP bus = opBusNext(ui, NULL);
93  thisApp = opBusApplicationLoad(
94  bus, options.application, OP_LDR_VERBOSE|OP_LDR_FAIL_IS_ERROR, 0
95  );
96 
97  }
98 
99  // Must advance to next phase for the API calls that follow
100  opRootModulePreSimulate(mi);
101 
102  if (thisApp) {
103  // Loaded on bus or memory so set start address manually
104  opProcessorPCSet(proc, opApplicationEntry(thisApp)); // set start address to application entry
105  }
106 
107  // use a custom argument to control if we analyze the application programs loaded on the instance
108  if ( options.analyzeinstances) {
109 
110  optObjectP object;
111 
112  if ( options.loadmemory) {
113  object = opObjectByName(
114  mi, MODULE_INSTANCE "/" MEM_INSTANCE, OP_MEMORY_EN
115  );
116  } else if ( options.loadbus) {
117  object = opObjectByName(
118  mi, MODULE_INSTANCE "/" BUS_INSTANCE, OP_BUS_EN
119  );
120  } else {
121  object = opObjectByName(
122  mi, MODULE_INSTANCE "/" CPU_INSTANCE, OP_PROCESSOR_EN
123  );
124  }
125 
126  opMessage("I", MODULE_INSTANCE, "Analyzing Loaded Applications");
127  optApplicationP thisApp = NULL;
128  while ((thisApp=opObjectApplicationNext(object, thisApp))) {
129 
130  optLoaderControls controls = opApplicationControls(thisApp);
131  Uns32 elfCode = opApplicationElfCode(thisApp);
132  optEndian endian = opApplicationEndian(thisApp);
133  Addr entry = opApplicationEntry(thisApp);
134  const char *path = opApplicationPath(thisApp);
135 
136  opMessage(
137  "I", MODULE_INSTANCE,
138  "%s : controls %x: elf 0x%x: endian %u: entry 0x" FMT_Ax ": path '%s'\n",
139  opObjectHierName(object),
140  controls,
141  elfCode,
142  endian,
143  entry,
144  path
145  );
146  }
147  }
148 
149  opRootModuleSimulate(mi);
150 
151  opSessionTerminate();
152 
153  return (opErrors() ? 1 : 0); // set exit based upon any errors
154 }
