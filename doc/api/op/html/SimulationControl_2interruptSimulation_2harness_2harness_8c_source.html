43 // LINUX signal handler to interrupt the running simulation

45 static void ctrlCHandler(Int32 nativeSigNum, siginfo_t *sigInfo, void *context) {
46  // On Linux the API call may be used directly as the simulator is frozen on a signal
47  // On Windows the simulator must be suspended before issuing this call

49  printf("External Users Ctrl-C Handler Triggered\n");
50  opInterrupt();
51 }

55 //
56 static void installCtrlCHandler(void) {
57 
58  struct sigaction sa = {{0}};
59  sa.sa_sigaction = ctrlCHandler;
60  sa.sa_flags = SA_SIGINFO;
61  sigfillset(&sa.sa_mask);
62  sigaction(SIGINT, &sa, NULL);
63 }

108  optProcessorP stoppedProcessor;

110  // simulate until done, yield or ctrl-C
111  Bool continueExecution = True;
112  while(continueExecution) {
113 
114  stoppedProcessor = opRootModuleSimulate(mr);
115 
116  if(stoppedProcessor) {
117  optStopReason stopReason = opProcessorStopReason(stoppedProcessor);
118 
119  switch(stopReason) {
120  case OP_SR_INTERRUPT:
121  case OP_SR_YIELD:
122  opMessage( "I", "HARNESS",
123  "%s: interrupt after " FMT_64u " instructions '%s'",
124  opObjectName(stoppedProcessor),
125  opProcessorICount(stoppedProcessor),
126  opStopReasonString(stopReason));
127  break;
128  default:
129  // some other stop reason encountered
130  continueExecution = False;
131  break;
132  }
133  } else {
134  // The simulation can stop without a processor handle
135  // i.e when all processors are finished
136  // use module stop reason
137  optStopReason stopReason = opRootModuleStopReason(mr);
138  opMessage( "I", "HARNESS",
139  "Simulation stopped '%s'", opStopReasonString(stopReason));
140  continueExecution = False;
141  }
142  }
143 
144  // print number of instructions executed at end of simulation and reason simulation stopped
145  opPrintf("Simulation finished '%s'\n", opStopReasonString(opRootModuleStopReason(mr)));
146  opPrintf("instructions executed cpu1 "FMT_64u" and cpu2 "FMT_64u"\n",
147  opProcessorICount(opObjectByName(mi, "cpu1", OP_PROCESSOR_EN).Processor),
148  opProcessorICount(opObjectByName(mi, "cpu2", OP_PROCESSOR_EN).Processor)
149  );
150 
151  opSessionTerminate();
152 
153  return (opErrors() ? 1 : 0); // set exit based upon any errors
