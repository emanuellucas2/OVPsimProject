26 int main(int argc, const char *argv[]) {
27  opSessionInit(OP_VERSION);
28 
29  opCmdParseStd(argv[0], OP_AC_ALL, argc, argv);
30 
31  optModuleP mr = opRootModuleNew(0, "top", 0);
32 
33  optModuleP mi = opModuleNew(
34  mr, // parent module
35  "module", // modelfile
36  "m1", // name
37  0,
38  0
39  );
40 
41  opRootModulePreSimulate(mr);
42 
43  // get the handle for the processor in the module
44  optPeripheralP uart1 = opObjectByName(mi, "uart1", OP_PERIPHERAL_EN).Peripheral;
45 
46  if(uart1) {
47  optBusPortP busPort = opObjectByName(uart1, "bport1", OP_BUSPORT_EN).BusPort;
48 #ifdef REPORT_ALL_REGISTERS
49  optMMRegisterP reg;
50  while((reg = opBusPortMMRegisterNext(busPort, reg))) {
51  opMessage("I", "HAR_BP", "Found register %s\n", opObjectName(reg));
52  }
53 #endif
54  optMMRegisterP msr = opObjectByName(busPort, "ab_msr", OP_MMREGISTER_EN).MMRegister;
55  if(msr) {
56 
57  // This is how to view a peripiheral register without changing the state of the peripheral
58  Uns8 v = 0;
59  opMMRegisterView(msr, &v);
60  opMessage("I", "HAR_RV", "ab_msr value=0x%x", v);
61 
62  v = 0xff;
63  opMessage("I", "HAR_WR", "writing ab_msr value=0x%x", v);
64 
65  // This is how to write to a peripheral register.
66  // Note that this will possibly change the state of the peripheral.
67  // The same effect can be achieved in OP by writing to the correct address on the appropriate bus.
68  opMMRegisterWrite(msr, &v);
69 
70  // This is how to view to a peripheral register.
71  // Note that if the peripheral register has a view function, or it otherwise correctly designed,
72  // this should not affect the state of the peripheral.
73  opMMRegisterView(msr, &v);
74  opMessage("I", "HAR_RV", "ab_msr VIEW value=0x%x", v);
75  } else {
76  opMessage("E", "HAR_NR", "Found no register called ab_msr");
77  }
78 
79  optMMRegisterP iir = opObjectByName(busPort, "ab_iir", OP_MMREGISTER_EN).MMRegister;
80  if(iir) {
81  Uns8 v = 0;
82  // This is how to read to a peripheral register.
83  // Note that this will possibly change the state of the peripheral.
84  opMMRegisterRead(iir, &v);
85  opMessage("I", "HAR_RD", "ab_iir READ value=0x%x", v);
86  opMMRegisterView(iir, &v);
87  opMessage("I", "HAR_RV", "ab_iir READ value=0x%x", v);
88  }
89 
90  } else {
91  opMessage("E", "HAR_NP", "Found no peripheral called uart1");
92  }
93  opSessionTerminate();
94 
95  return (opErrors() ? 1 : 0); // set exit based upon any errors
96 }
