91 static int loadBinaryFile(optMemoryP memory, const char *fileName, Uns32 baseAddress) {
92 
93  FILE *fp;
94  char inBuf[MAX_LINE_LENGTH + 1];
95  int address = baseAddress;
96 
97  fp = fopen(fileName, "rb");
98 
99  if (!fp) {
100  opMessage("E", HARNESS_NAME "_BIN_LDR",
101  "Failed to open Memory Initialization File '%s'",
102  fileName);
103  return -1;
104  }
105 
106  if(DIAG_LOW)
107  opMessage("I", HARNESS_NAME "_BIN_LDR",
108  "Loading binary file '%s'",
109  fileName);
110 
111  Uns32 bytesRead = 0;
112  while ( (bytesRead = fread( inBuf, 1, MAX_LINE_LENGTH, fp)) != 0 ) {
113  //
114  // Access the memory
115  //
116  if(!opMemoryWrite(
117  memory, // memory instance
118  address, // address offset into memory
119  inBuf, // bytes to write
120  bytesRead, // number of bytes to write
121  True // artifact access
122  )) {
123  opMessage("E", HARNESS_NAME "_BIN_LDR_FW",
124  "Failed Data Write at 0x%08x",
125  address);
126  return -1;
127  }
128 
129  if(DIAG_MEDIUM)
130  opMessage("I", HARNESS_NAME "_BIN_LDR_LD",
131  "Load address 0x%08x %u bytes",
132  address, bytesRead);
133  address += bytesRead;
134  }
135 
136  if(DIAG_LOW)
137  opMessage("I", HARNESS_NAME "_BIN_LDR",
138  "Load Complete");
139 
140  if (fclose(fp)!=0) {
141  opMessage("E", HARNESS_NAME "_BIN_LDR_CF",
142  "Failed to close Memory Initialization File");
143  return -1;
144  }
145 
146 
147  return (opErrors() ? 1 : 0); // set exit based upon any errors
148 }
