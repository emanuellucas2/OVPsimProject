50 // Give the monitoring a threshold
51 static unsigned int count;
52 #define MONITOR_THRESHOLD 100
53 
54 //
55 // triggered when registered access happens and prints information of access
56 //
57 static OP_MONITOR_FN(monitorCallback) {
58  if(++count <= MONITOR_THRESHOLD) {
59  opMessage ("I", PREFIX "_MT",
60  "Monitor triggered (%03d): "
61  "callback '%s': '%s' : "
62  "type '%s' : bytes %u : "
63  "address Physical 0x" FMT_A0Nx" Virtual 0x" FMT_A0Nx " %s",
64  count,
65  __FUNCTION__,
66  processor ? opObjectName(processor) : "artifact", // if no processor this is an artifact access
67  (const char*)userData,
68  bytes,
69  addr,
70  VA,
71  (count == MONITOR_THRESHOLD) ? "(threshold reached)" : ""
72  );
73  }
74 }

76 //
77 // iterate across the processors found in the module and register callbacks for read, write and fetch
78 //
79 static void monitorProcessor(optModuleP mi) {
80 
81  optModuleP mod;
82 
83  if (!(mod = opObjectByName (mi, MODULE_INSTANCE, OP_MODULE_EN).Module)) {
84  opMessage ("F", PREFIX "_NFW", "Can not find module(%s)", MODULE_INSTANCE);
85  }
86 
87  // iterate across all processors found in module
88  optProcessorP processor = 0;
89  while ((processor = opProcessorNext(mod, processor))) {
90 
91  Addr max = 0;
92  optBusPortConnP bpc = opObjectByName(processor, "DATA", OP_BUSPORTCONN_EN).BusPortConn;
93  if(bpc) {
94  optBusP bus = opBusPortConnBus(bpc);
95  max = opBusMaxAddress(bus);
96  }
97  Addr min = 0;
98 
99  opMessage("I", PREFIX "_BM", "Add monitor for '%s' (0x" FMT_A0Nx " to 0x" FMT_A0Nx ")\n",
100  opObjectHierName(processor), (Addr)0, max);
101  opProcessorFetchMonitorAdd(processor, min, max, monitorCallback, "processor-fetch");
102  opProcessorReadMonitorAdd (processor, min, max, monitorCallback, "processor-read");
103  opProcessorWriteMonitorAdd(processor, min, max, monitorCallback, "processor-write");
104 
105  }
106 }

174 // Main

176 int main(int argc, const char *argv[]) {
177 
178  opSessionInit(OP_VERSION);
179 
180  // Command line parser
181  optCmdParserP parser = opCmdParserNew(HARNESS_NAME, OP_AC_ALL);
182  cmdParserAddArgs(parser);
183  opCmdParseArgs(parser, argc, argv);
184 
185  // instance module
186  optModuleP rm = opRootModuleNew(0, 0, 0);
187  opModuleNew(
188  rm, // parent module
189  MODULE_DIR, // modelfile
190  MODULE_INSTANCE, // name
191  0,
192  0
193  );
194 
195  // setup for simulation i.e. construction complete
196  opRootModulePreSimulate(rm);
197 
198  if (options.moduleshow) {
199 
200  // Show module(s) contents
201  opModuleShow (rm);
202 
203  } else if (options.busshow) {
204 
205  // find and show bus connections
206  monitorBus(rm);
207 
208  } else {
209 
210  // Setup processor monitors
211  monitorProcessor(rm);
212 
213  // Setup bus monitors
214  monitorBus(rm);
215 
216  // Setup memory monitors
217  monitorMemory(rm);
218 
219  // simulate root module
220  opRootModuleSimulate(rm);
221 
222  }
223 
224  // terminate
225  opSessionTerminate();
226 
227  return (opErrors() ? 1 : 0); // set exit based upon any errors
228 
229 }
