30 int main(int argc, const char *argv[]) {
31  opSessionInit(OP_VERSION);
32 
33  opCmdParseStd (argv[0], OP_AC_ALL, argc, argv);
34 
35  // root module
36  optModuleP mi = opRootModuleNew(0, 0, 0);
37 
38 #ifdef PARAMLIST
39  opMessage("I", "HARNESS", "Define parameters using optParamP parameter list");
40  optParamP paramList = NULL;
41 
42 #ifdef TOPLEVEL
43  // Set formal parameter on the top level module to change uart addresses in submodule
44  opMessage("I", "HARNESS", "Set formal parameters on topmodule");
45  paramList = opParamUns64Set(paramList, "subUart0Address", 0x100003e8);
46 #ifdef DUALBASE
47  // Two separate base addresses at top level
48  paramList = opParamUns64Set(paramList, "subUart1Address", 0x100003f8);
49 #endif
50 #else /* !TOPLEVEL */
51  // Set formal parameter directly on the sub level module to change uart addresses on UARTs
52  opMessage("I", "HARNESS", "Set formal parameters on submodule");
53  paramList = opParamUns64Set(paramList, "subUart0/baseAddress", 0x100003e8);
54  paramList = opParamUns64Set(paramList, "subUart1/baseAddress", 0x100003f8);
55 #endif
56 
57  // Set parameter on instance of uarts to set logfiles
58  paramList = opParamStringSet(paramList, "subUart0/uart0/outfile", "subUart0-uart0.txt");
59  paramList = opParamStringSet(paramList, "subUart1/uart0/outfile", "subUart1-uart0.txt");
60 
61  // instance the topmodule into the root module
62  opModuleNew(mi, "./topmodule", "top", 0, paramList);
63 
64 #else /* !PARAMLIST */
65 
66  opMessage("I", "HARNESS", "Define parameters using OP_PARAMS Macro list");
67  // instance the topmodule into the root module
68  opModuleNew(mi, "./topmodule", "top", 0,
69  OP_PARAMS (
70  OP_PARAM_UNS64_SET ("subUart0/baseAddress", 0x100003e8),
71  OP_PARAM_UNS64_SET ("subUart1/baseAddress", 0x100003f8),
72  OP_PARAM_STRING_SET ("subUart0/uart0/outfile", "subUart0-uart0.txt"),
73  OP_PARAM_STRING_SET ("subUart1/uart0/outfile", "subUart1-uart0.txt")
74  )
75  );
76 #endif
77 
78  // run simulation
79  opRootModuleSimulate(mi);
80 
81  opSessionTerminate();
82 
83  return (opErrors() ? 1 : 0); // set exit based upon any errors
84 }
