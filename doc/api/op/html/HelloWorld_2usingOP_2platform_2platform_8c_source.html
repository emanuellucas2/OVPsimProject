35 static OP_CONSTRUCT_FN(instantiateComponents) {
36 
37 
38  // Processor model and semihosting, selected from VLNV library below
39  const char *model = NULL;
40  const char *semihosting = NULL;
41 
42  if (strcmp(options.processortype, "riscv") == 0) {
43  model = opVLNVString(0, options.alternatevendor ? options.alternatevendor :
44  "riscv.ovpworld.org", "processor", "riscv", "1.0", OP_EXTENSION, OP_VLNV_QUIET);
45  semihosting = opVLNVString(0, "riscv.ovpworld.org", "semihosting", "pk", "1.0", OP_EXTENSION, OP_VLNV_QUIET);
46 
47  } else if (strcmp(options.processortype, "mips32") == 0) {
48  model = opVLNVString(0, options.alternatevendor ? options.alternatevendor :
49  "mips.ovpworld.org", "processor", "mips32_r1r5", "1.0", OP_EXTENSION, OP_VLNV_QUIET);
50  semihosting = opVLNVString(0, "mips.ovpworld.org", "semihosting", "mips32Newlib", "1.0", OP_EXTENSION, OP_VLNV_QUIET);
51 
52  } else if (strcmp(options.processortype, "arm") == 0) {
53  model = opVLNVString(0,options.alternatevendor ? options.alternatevendor :
54  "arm.ovpworld.org", "processor", "arm", "1.0", OP_EXTENSION, OP_VLNV_QUIET);
55  semihosting = opVLNVString(0, "arm.ovpworld.org", "semihosting", "armNewlib", "1.0", OP_EXTENSION, OP_VLNV_QUIET);
56 
57  } else {
58  opMessage("W", "PROC", "Processor type '%s' not recognized. "
59  "Please try 'riscv', 'mips32' or 'arm'", options.processortype);
60  }
61 
62  if (!model || !semihosting) {
63  opMessage("F", "LIBRARY", "Failed to find model or semihost library components"
64  " for processor type '%s ' (or vendor '%s')",
65  options.processortype, options.alternatevendor);
66  }
67 
68 
69  // Bus bus
70  optBusP bus_b = opBusNew(mi, "bus", 32, 0, 0);
71 
72  // Processor CPU1
73  optProcessorP CPU1_c = opProcessorNew(
74  mi,
75  model,
76  "CPU1",
77  OP_CONNECTIONS(
78  OP_BUS_CONNECTIONS(
79  OP_BUS_CONNECT(bus_b, "INSTRUCTION"),
80  OP_BUS_CONNECT(bus_b, "DATA")
81  )
82  ),
83  0
84  );
85 
86  opProcessorExtensionNew(CPU1_c, semihosting, "semihost", 0 );
87 
88  // Memory
89  opMemoryNew(
90  mi,
91  "memory",
92  OP_PRIV_RWX,
93  (0xffffffffULL) - (0x0ULL),
94  OP_CONNECTIONS(
95  OP_BUS_CONNECTIONS(
96  OP_BUS_CONNECT(bus_b, "sp1", .slave=1, .addrLo=0x0ULL, .addrHi=0xffffffffULL)
97  )
98  ),
99  0
100  );
101 
102 }
