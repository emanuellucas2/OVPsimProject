79 static int loadHexFile(optProcessorP processor, const char *fileName) {
80 
81  FILE *fp;
82  char inBuf[MAX_LINE_LENGTH + 1];
83  int address, endAddress, data, dataCheck;
84 
85  fp = fopen(fileName, "r");
86 
87  if (!fp) {
88  opMessage("E", HARNESS_NAME "_HEX_LDR",
89  "Failed to open Memory Initialization File '%s'",
90  fileName);
91  return -1;
92  }
93 
94  opMessage("I", HARNESS_NAME "_HEX_LDR",
95  "Loading Hex file '%s'",
96  fileName);
97 
98  while ( fgets( inBuf,MAX_LINE_LENGTH, fp) != 0 ) {
99 
100  if ( inBuf[0] == '#') {
101  // ignore header
102  } else if ( inBuf[0] == ':') {
103  // symbol
104  Uns32 symbolAddress;
105  char symbol[16];
106  if ( sscanf(inBuf, ":%08x %s", &symbolAddress, symbol) == 2 ) {
107  opMessage("I", HARNESS_NAME "_HEX_LDR_SYM", "Found symbol '%s' at address 0x%08x",
108  symbol,
109  symbolAddress);
110 
111  const char *runTime=getenv("IMPERAS_RUNTIME");
112  if(runTime && strcmp(runTime, "CpuManager") == 0) {
113  opProcessorApplicationSymbolAdd(processor,
114  symbol,
115  symbolAddress,
116  4,
117  ORD_SYMBOL_TYPE_FUNC,
118  ORD_SYMBOL_BIND_GLOBAL);
119  } else {
120  opMessage("W", HARNESS_NAME "_HEX_LDR_SF",
121  "Symbols cannot be loaded with this product runtime (%s), "
122  "supported in Imperas Professional simulator",
123  runTime);
124  }
125  if(strcmp(symbol, "_start") == 0) {
126  opMessage("I", HARNESS_NAME "_HEX_LDR_START",
127  "Set start address to 0x%08x",
128  symbolAddress);
129  opProcessorPCSet(processor, symbolAddress);
130  }
131 
132  } else {
133  opMessage("W", HARNESS_NAME "_HEX_LDR_SND",
134  "Found symbol line '%s' but not decoded",
135  inBuf);
136  }
137 
138  } else {
139  if ( sscanf(inBuf, "%08x-%08x %08x ;", &address, &endAddress, &data) != 3 ) {
140  sscanf(inBuf, "%08x %08x ;", &address, &data);
141  endAddress = address;
142  }
143 
144  do {
145  //
146  // Access the memory through the processor memory space
147  //
148  if(!opProcessorWrite(
149  processor, // processor
150  address, // memory address
151  &data, // data buffer of data to write
152  4, // number of bytes in one object
153  1, // number of objects
154  True, // debug access (not a true processor access)
155  OP_HOSTENDIAN_TARGET
156  )) {
157  opMessage("E", HARNESS_NAME "_HEX_LDR_FW",
158  "Failed Data Write at 0x%08x",
159  (Uns32)address);
160  return -1;
161  }
162 
163  opProcessorRead(
164  processor,
165  address,
166  &dataCheck,
167  4,
168  1,
169  True,
170  OP_HOSTENDIAN_TARGET
171  );
172 
173  if(data != dataCheck) {
174  opMessage("E", HARNESS_NAME "_HEX_LDR_FRB",
175  "Failed Data Read Back at 0x%08x (0x%08x 0x%08x)",
176  (Uns32)address, data, dataCheck);
177  return -1;
178  }
179 
180  opMessage("I", HARNESS_NAME "_HEX_LDR_LD",
181  "Load address 0x%08x : 0x%08x",
182  address, data);
183 
184  } while ( address++ < endAddress);
185  }
186  }
187 
188  opMessage("I", HARNESS_NAME "_HEX_LDR",
189  "Load Complete");
190 
191  if (fclose(fp)!=0) {
192  opMessage("E", HARNESS_NAME "_HEX_LDR_CF",
193  "Failed to close Memory Initialization File");
194  return -1;
195  }
196 
197 
198  return (opErrors() ? 1 : 0); // set exit based upon any errors
199 }
