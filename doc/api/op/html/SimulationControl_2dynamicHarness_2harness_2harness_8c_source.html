90 int main(int argc, const char **argv) {
91 
92  checkEnvironment(argv[0]);
93 
94  opSessionInit(OP_VERSION);
95  optCmdParserP parser = opCmdParserNew(argv[0], OP_AC_SIM | OP_AC_LIB);
96 
97  opCmdUsageMessage(parser, helpMessage);
98 
99  modelSpec model = { .vlnv={0, "module", 0, "1.0"}, .file=0, .parameters=NULL, .instcount=1};
100 
101  opCmdParserAdd (parser, "modulevendor", "V", 0, "harness", OP_FT_STRINGVAL, &model.vlnv.vendor, "Module vendor", OP_AC_ALL, 0, 0);
102  opCmdParserAdd (parser, "modulelibrary", "L", 0, "harness", OP_FT_STRINGVAL, &model.vlnv.library, "Module library (defaults to 'module')", OP_AC_ALL, 0, 0);
103  opCmdParserAdd (parser, "modulename", "N", 0, "harness", OP_FT_STRINGVAL, &model.vlnv.name, "Module name", OP_AC_ALL, 0, 0);
104  opCmdParserAdd (parser, "moduleversion", "S", 0, "harness", OP_FT_STRINGVAL, &model.vlnv.version, "Module version (defaults to '1.0')" , OP_AC_ALL, 0, 0);
105  opCmdParserAdd (parser, "modulefile", 0, 0, "harness", OP_FT_STRINGVAL, &model.file, "Path to a a module" , OP_AC_ALL, 0, 0);
106  opCmdParserAdd (parser, "parameter", 0, 0, "harness", OP_FT_PAIRLIST, &model.parameters, "Module model parameters" , OP_AC_ALL, 0, 0);
107  opCmdParserAdd (parser, "instancecount", 0, 0, "harness", OP_FT_UNS32VAL, &model.instcount, "Number of instances of the module" , OP_AC_ALL, 0, 0);
108 
109  Bool ok = opCmdParseArgs(parser, argc, argv);
110  if(!ok) {
111  exitWithErrors();
112  }
113 
114  const char *path;
115 
116  if(!model.vlnv.name && !model.file) {
117  opMessage("E", "HAR_NPRC", "No module was specified. Suggest --modulename <module> or --modulefile <filename>");
118  opPrintf("%s", helpMessage);
119  exitWithErrors();
120  }
121 
122  if(model.vlnv.name) {
123  path = opVLNVString(
124  0, // use the default VLNV path
125  model.vlnv.vendor,
126  model.vlnv.library,
127  model.vlnv.name,
128  model.vlnv.version,
129  OP_MODULE,
130  OP_VLNV_FATAL
131  );
132 
133  } else {
134  path = model.file;
135  }
136 
137  optModuleP mi = opRootModuleNew(0, 0, 0);
138 
139  // Create number of instances of the module specified on the command line
140  Uns32 i;
141  for (i = 0; i < model.instcount; i++) {
142 
143  // Create unique name for this instance
144  char instName[16];
145  snprintf(instName, sizeof(instName), "m%d", i);
146 
147  // Create parameter list with any parameters specified on command line
148  // (Note: unique parameter list required for each opModuleNew call)
149  optParamP paramList = NULL;
150  optArgStringListP p;
151  for (p = model.parameters; p; p= p->next) {
152  paramList = opParamStringSet(paramList, p->name, p->value);
153  }
154 
155  if (!opModuleNew(mi, path, instName, 0, paramList)) {
156  exitWithErrors();
157  }
158  }
159 
160  opRootModuleSimulate(mi);
161  opSessionTerminate();
162 
163  return (opErrors() ? 1 : 0); // set exit based upon any errors
164 }
