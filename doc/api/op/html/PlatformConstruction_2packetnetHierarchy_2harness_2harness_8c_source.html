70 //
71 // Send a packetnet and reports the fact
72 //
73 static void packetWrite(optModuleP mi) {
74  optPacketnetP pkn = opObjectByName(mi, "pktnet", OP_PACKETNET_EN).Packetnet;
75 
76  netPacket p = { 1, {'I', 'C', 'M', 0, 0, 0 }, 0 };
77 
78  opMessage("I", HARNESS_NAME "_PKT_TXS",
79  "%s: writePkt START {%02x}, {%s}, {%02x}",
80  opObjectHierName(pkn),
81  p.head,
82  p.data,
83  p.tail
84  );
85 
86  // This is the function that triggers the packetnet.
87  // All models with a callback installed on this packetnet
88  // will be notified while this function is active.
89 
90  opPacketnetWrite(pkn, &p, sizeof(netPacket));
91 
92  // Now that the function has returned, we are certain that all
93  // models have been notified. Their notifier callbacks have finished,
94  // and if thet chose to modify the data in the packet, we can see the
95  // modification here.
96 
97  opMessage("I", HARNESS_NAME "_PKT_TXE",
98  "%s: writePkt DONE {%02x}, {%s}, {%02x}",
99  opObjectHierName(pkn),
100  p.head,
101  p.data,
102  p.tail
103  );
104  opPrintf("\n");
105 }
