186 // Command type : describes how the command is used
187 #define VMI_CT_MASK 0x0000000F // Bits used for the command type
188 #define VMI_CT_QUERY 0x00000001 // Command used to query something
189 #define VMI_CT_STATUS 0x00000002 // Called when the simulator stops
190 #define VMI_CT_MODE 0x00000003 // Command that changes simulator behavior
191 
192 // Command object : describes what the command operates on
193 #define VMI_CO_MASK 0x00000FF0 // Bits used for the command group
194 #define VMI_CO_CACHE 0x00000010 // cache
195 #define VMI_CO_CPU 0x00000020 // processor
196 #define VMI_CO_CONTEXT 0x00000030 // stack trace
197 #define VMI_CO_DIAG 0x00000040 // diagnostics
198 #define VMI_CO_FUNCTION 0x00000050 // Function
199 #define VMI_CO_GIC 0x00000060 // GIC
200 #define VMI_CO_LINE 0x00000070 // code source lines
201 #define VMI_CO_LKM 0x00000080 // Loadable kernel module
202 #define VMI_CO_MEMORY 0x00000090 // Memory
203 #define VMI_CO_OS 0x000000A0 // OS or scheduler
