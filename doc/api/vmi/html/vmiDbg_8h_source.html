243 //
244 // Structure used to pass information about the registers in a simulated
245 // processor to the simulator
246 //
247 typedef struct vmiRegInfoS {
248  const char *name; // register identification name
249  const char *description; // description string
250  vmiRegGroupCP group; // group for this register
251  Uns32 gdbIndex; // gdb ordinal index number
252  vmiRegUsage usage : 3; // any special usage for this register
253  vmiRegAccess access : 2; // allowed access
254  Bool noSaveRestore : 1; // does not participate in save/restore
255  Bool noTraceChange : 1; // does not participate in tracing
256  Bool instrAttrIgnore : 1; // ignore for instruction attributes
257  Bool isAlias : 1; // register an alias of another?
258  Bool extension : 1; // defined in an extension library?
259  Uns32 unused : 22; // unused fields
260  Uns32 bits; // size of register in bits
261  vmiRegReadFn readCB; // read callback function
262  vmiRegWriteFn writeCB; // write callback function
263  vmiReg raw; // raw register value (if no callback)
264  void *userData; // model-specific data pointer
265 } vmiRegInfo;

267 //
268 // Special readCB and writeCB values specifying that read and write of raw
269 // register values is allowed (i.e. without callback), requiring that either:
270 // 1. the 'raw' field is set with the appropriate register; or:
271 // 2. the userData argument is set to the register offset in the processor
272 // structure (deprecated backwards-compatible mode).
273 // With the introduction of the 'raw' field in the vmiRegInfo structure, these
274 // special values are not required (specify the 'raw' vmiReg and omit the the
275 // read/write callback to achieve the same effect).
276 //
277 #define VMI_REG_RAW_READ_CB ((vmiRegReadFn)(-1))
278 #define VMI_REG_RAW_WRITE_CB ((vmiRegWriteFn)(-1))
