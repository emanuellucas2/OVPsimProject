121 //
122 // The vmiosInterceptDesc structure describes a single intercept. If 'name' is
123 // non-NULL, the intercept will apply when the processor executes at that
124 // symbol; otherwise, the intercept applies when the processor executes at
125 // virtual address 'simAddress'. The intercept can be either raw or abstract,
126 // as described below.
127 //
128 // RAW INTERCEPT
129 // -------------
130 // For a raw intercept, 'interceptCB' is non-NULL and specifies a function of
131 // type vmiosInterceptFn that is called when executing at the intercept address,
132 // as described above. The callback will typically use calls to vmirtRegRead and
133 // vmirtRegWrite to directly access processor registers, and will therefore be
134 // processor-specific.
135 //
136 // ABI INTERCEPT
137 // -------------
138 // For an ABI (application binary interface) intercept, 'abiCB' and 'format'
139 // must be non-NULL. The 'format' string is a list of characters describing
140 // arguments that are supplied in order to 'abiCB', as follows, with the
141 // corresponding type required for the intercept function parameter:
142 //
143 // 'p': the current processor vmiProcessorP
144 // 'o': the current intercept object vmiosObjectP
145 // 'c': name of the intercepted function const char *
146 // 'b': Boolean argument using standard ABI Bool
147 // '4': 4-byte argument using standard ABI Uns32 or Int32
148 // '8': 8-byte argument using standard ABI Uns64 or Int64
149 // 'a': pointer argument using standard ABI Addr
150 // 'd': double argument using standard ABI double
151 //
152 // 'p', 'o' and 'c' arguments are implicit and must precede all standard ABI
153 // arguments; all others are passed from the intercepted application function
154 // using the standard processor ABI. If 'returnType' is non-zero, it specifies
155 // the function return type using the standard application binary interface. It
156 // must be one of 'b', '4', '8', 'a' or 'd'.
157 //
158 // As an example, this function in application code:
159 //
160 // Uns32 testFunc(void *a, double b, Uns64 c) {}
161 //
162 // could be intercepted using this ABI intercept description:
163 //
164 // Name Address Opaque Callback
165 // {"testFunc", 0, True, VMIOS_ABI_INT('4', "poad8", testFuncInt)}
166 //
167 // with the host code ABI intercept function defined as:
168 //
169 // static Uns32 testFuncInt(
170 // vmiProcessorP processor,
171 // vmiosObjectP object,
172 // Addr a,
173 // double b,
174 // Uns64 c
175 // );
176 //
177 // ABI INTERCEPT RESTRICTIONS
178 // --------------------------
179 // 1. ABI intercepts are supported for PSEs, but not all processor models
180 // currently support them;
181 // 2. A maximum of 16 function parameters is supported;
182 // 3. Threaded intercepts are not supported.
183 //
184 typedef struct vmiosInterceptDescS {
185 
186  // INTERCEPT DESCRIPTION
187  const char *name; // for interception by name
188  Addr simAddress; // for interception by address
189  vmiosInterceptAttr attrs; // intercepted function attributes
190 
191  // RAW INTERCEPT SUPPORT
192  vmiosInterceptFn interceptCB; // interception callback
193  void *userData; // client-specific data pointer
194 
195  // ABI INTERCEPT SUPPORT
196  vmiCallFn abiCB; // ABI callback
197  const char *format; // arguments to ABI callback
198  char returnType; // return type from ABI callback
199 
200 } vmiosInterceptDesc;

215 //
216 // This routine may be used to override the default behavior of the morpher for
217 // a particular instruction. If the instruction should be overridden, the
218 // function should return a vmiosInterceptFn function for the required
219 // interception and set byref arguments 'opaque' (True->opaque, False->
220 // transparent) 'context' (a user-defined context string) and 'userData'
221 // (an application-specific argument passed uninterpreted to the intercept
222 // function); otherwise, it should return NULL.
223 //
224 #define VMIOS_MORPH_FN(_NAME) vmiosInterceptFn _NAME( \
225  vmiProcessorP processor, \

227  Addr thisPC, \

295 //
296 // This function is called when the 'next' or 'step' command is issued
297 // to the MPD. It has two purposes:
298 // 1. for the debugger to determine if the operation is supported.
299 // 2. for the intercept library to store the current context.
300 //
301 // It should return true if the operation can be supported, false otherwise.
302 // Typically, for 'step' (step into) it should store the current line and file.
303 // For 'next' (step over) it should also store any registers required to
304 // determine when the program has reached the original call frame.
305 //
306 #define VMIOS_STEP_LINE_BEGIN_FN(_NAME) Bool _NAME ( \
307  vmiProcessorP processor, \
308  vmiosObjectP object, \
309  vmiStepOperation operation, \
310  const char *currentFile, \
311  Uns32 currentLine \
312 )
313 typedef VMIOS_STEP_LINE_BEGIN_FN((*vmiosStepLineBeginFn));
314 
315 //
316 // This function is called when the 'next' or 'step' command is issued, but
317 // after each instruction step. It should return True if more instruction steps
318 // are required, False if the program has reached the desired location.
319 // Typically, for 'step' (step into) it should compare the current line and file
320 // with the stored values.
321 // For 'next' (step over) it should also compare any registers required to
322 // determine when the program has reached the original call frame.
323 //
324 // These functions will depend on the ABI used by the current program.
325 //
326 #define VMIOS_STEP_LINE_ITERATE_FN(_NAME) Bool _NAME ( \
327  vmiProcessorP processor, \
328  vmiosObjectP object, \
329  vmiStepOperation operation, \
330  const char *currentFile, \
331  Uns32 currentLine \
332 )
333 typedef VMIOS_STEP_LINE_ITERATE_FN((*vmiosStepLineIterateFn));
