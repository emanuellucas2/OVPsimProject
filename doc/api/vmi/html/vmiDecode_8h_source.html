25 //
26 // Add a new entry to the decode table. 'mask' is a bitmask which is applied to
27 // a candidate instruction, and 'value' is compared with the masked value. If
28 // the masked value matches, 'matchValue' is returned.
29 //
30 // Sometimes it is required to create conflicting decode entries in a table
31 // (for example, a particular instruction variant may have a different
32 // mnemonic or special functionality different to the general case). In such
33 // cases, use the 'priority' argument to specify which of the conflicting
34 // decodes should be used (larger 'priority' value are chosen in preference)
35 //
36 Bool vmidNewEntry(
37  vmidDecodeTableP table,
38  const char *name,
39  Uns32 matchValue,
40  Uns64 mask,
41  Uns64 value,
42  Int32 priority
43 );
44 
45 //
46 // Add a new entry to the decode table using a format string, which describes
47 // the bit pattern of an opcode from MSB to LSB. The characters in the string
48 // are as follows:
49 //
50 // '1': corresponding bit must be 1
51 // '0': corresponding bit must be 0
52 // '|', ',', '/', ' ', tab: ignored
53 // any other character: corresponding bit can be either 0 or 1
54 //
55 // As an example, this format string specifies a 16-bit opcode with the
56 // most-significant 5 bits and least-significant 3 bits having fixed values,
57 // with other bit values unconstrained. Note that the spacer character '|' is
58 // ignored, and the value '.' is used in this case to denote an unconstrained
59 // bit value:
60 //
61 // "|10001|........|101|"
62 //
63 Bool vmidNewEntryFmtBin(
64  vmidDecodeTableP table,
65  const char *name,
66  Uns32 matchValue,
67  const char *format,
68  Int32 priority
69 );
