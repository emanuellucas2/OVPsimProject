118 //
119 // This callback function type defines functions that are called just before
120 // a code block translation starts (startBlockCB) and just after a block
121 // translation completes (endBlockCB). The callbacks may be used to initialize
122 // and finalize data structures required by the translation function below.
123 // 'thisPC' and 'inDelaySlot' arguments have the same meanings as for the code
124 // translation function of type vmiMorphFn, documented below. 'blockState'
125 // is passed a pointer to a model-defined structure of size given by the
126 // 'blockStateSize' field in the vmiIASAttr structure definition for the
127 // processor; the 'blockState' structure is initialized to zero and should be
128 // used to hold state required while the translation of a single code block
129 // is in progress. 'reason' indicates the reason for the translation: see the
130 // comment associated with the vmiMorphReason type definition for more
131 // information
132 //
133 #define VMI_START_END_BLOCK_FN(_NAME) void _NAME( \
134  vmiProcessorP processor, \
135  Addr thisPC, \
136  Uns8 inDelaySlot, \
137  vmiMorphReason reason, \
138  void *blockState \
139 )
140 typedef VMI_START_END_BLOCK_FN((*vmiStartEndBlockFn));

142 //
143 // This callback function type defines a function used to translate a single
144 // processor instruction at address 'thisPC'. On processors that support delay
145 // slots, a non-zero value of 'inDelaySlot' indicates that this is a delay
146 // slot instruction at the specified offset from the jump instruction for which
147 // this delay slot instruction applies. 'firstInBlock' is True if this
148 // instruction is the first in a code block. 'emitTrace' is True if the current
149 // code block has trace enabled.
150 //
151 // 'blockState' is passed a pointer to a model-defined structure of size given
152 // by the 'blockStateSize' field in the vmiIASAttr structure definition for the
153 // processor. The 'blockState' structure is initialized to zero at the start
154 // of each code block and should be used to hold state common to all
155 // instructions in a code block.
156 //
157 // 'instrState' is passed a pointer to a model-defined structure of size given
158 // by the 'instrStateSize' field in the vmiIASAttr structure definition for the
159 // processor. The 'instrState' structure is initialized to zero before each
160 // instruction and should be used to hold state for that instruction.
161 //
162 // Processors may define up to three functions of this type, with the following
163 // field names in the vmiIASAttr structure:
164 //
165 // preMorphCB
166 // ----------
167 // This function is called *before* an instruction is translated. Typically, it
168 // will initialize fields in the blockState and instrState structures for later
169 // use in the morphCB callback. The function is called whether or not the
170 // instruction is opaquely intercepted; it may be omitted (set to zero) if no
171 // special pre-instruction actions are required.
172 //
173 // morphCB
174 // -------
175 // This function implements behavior for the current instruction, unless it is
176 // opaquely intercepted. It should not be omitted.
177 //
178 // postMorphCB
179 // -----------
180 // This function is called *after* an instruction is translated. Typically, it
181 // will perform common actions required at the end of all instructions. The
182 // function is called whether or not the instruction is opaquely intercepted;
183 // it may be omitted (set to zero) if no special post-instruction actions are
184 // required.
185 //
186 #define VMI_MORPH_FN(_NAME) void _NAME( \
187  vmiProcessorP processor, \
188  Addr thisPC, \
189  Uns8 inDelaySlot, \
190  Bool firstInBlock, \
191  Bool emitTrace, \
192  void *blockState, \
193  void *instrState \
194 )
195 typedef VMI_MORPH_FN((*vmiMorphFn));
