75 //
76 // Called to enable/disable a cache and register/clear callbacks for cache events
77 //
78 // When enable is true the cache is enabled, when false it is disabled
79 //
80 // CB functions may be specified as NULL if no callback is required.
81 //
82 // Returns a unique name for the cache if successful, If the same cache
83 // is connected to multiple processors they will each get the same name returned.
84 // NULL if no cache with that type is available
85 //
86 // userData will be passed to callback functions
87 //
88 // To get callbacks on all cache events this must be called for every
89 // processor connected to the cache when the cache is shared between
90 // multiple processors (e.g. in a multi-threaded processor)
91 //
92 #define VMI_CACHE_REGISTER_FN(_NAME) const char * _NAME( \
93  vmiProcessorP processor, \
94  vmiCacheType cacheType, \
95  Bool enable, \
96  vmiCacheHitFn hitCB, \
97  vmiCacheMissFn missCB, \
98  vmiCacheInvalidateFn invalidateCB, \
99  void *userData \
100 )
101 typedef VMI_CACHE_REGISTER_FN((*vmiCacheRegisterFn));
