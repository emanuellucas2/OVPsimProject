58 //
59 // VMI_CPU_DBG - Utility macro for debug messages using processor
60 //
61 // Useful macro for controlling debug messages generated by a processor model.
62 // Define CPU_DEBUG_FLAG to identify which bits in the processor flags should
63 // cause debug messages to be generated.
64 //
65 #define VMI_CPU_DBG(_PROC, _M, _FMT, ...) \
66  do { \
67  if (vmirtProcessorFlags(_PROC) & CPU_DEBUG_FLAG) \
68  vmiMessage("I", _M, _FMT, ## __VA_ARGS__); \
69  } while (0)
