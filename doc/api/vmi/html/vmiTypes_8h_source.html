112 //
113 // This bitmask is used with vmirtRead*/vmirtWrite* functions to describe the
114 // access characteristics. Valid values are:
115 //
116 // MEM_AA_FALSE: indicates that the access is an artifact read/write, which
117 // should not update processor state.
118 //
119 // MEM_AA_TRUE: indicates that the access is a true read/write which may update
120 // processor state (for example, it may initiate entry to an exception handler
121 // to handle an illegal address).
122 //
123 // MEM_AA_FETCH: indicates that the access is an artifact fetch, which should
124 // not update processor state.
125 //
126 // MEM_AA_TRUE|MEM_AA_FETCH: indicates that the access is a true fetch which may
127 // update processor state (for example, it may initiate entry to an exception
128 // handler to handle an illegal address).
129 //
130 typedef enum memAccessAttrsE {
131  MEM_AA_FALSE = 0x0, // this is an artifact access
132  MEM_AA_TRUE = 0x1, // this is a true processor access
133  MEM_AA_FETCH = 0x2, // this access is a fetch
134 } memAccessAttrs;
