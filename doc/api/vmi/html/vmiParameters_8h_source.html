22 typedef enum vmiParameterTypeE {
23  vmi_PT_BOOL, // Boolean parameter. Can be 0,n,N,f,F,dis etc. or 1,y,Y,t,T,en
24  vmi_PT_INT32, // Signed integer parameter.
25  vmi_PT_INT64, // Signed integer parameter.
26  vmi_PT_UNS32, // Unsigned integer parameter.

28  vmi_PT_DOUBLE, // Floating point parameter.
29  vmi_PT_STRING, // String parameter.
30  vmi_PT_ENUM, // Enum parameter.
31  // Specified by a string from a set of strings.
32  // Returns an integer.
33  vmi_PT_ENDIAN, // Takes one of two values; little or big
34  vmi_PT_PTR, // Pointer parameter.
35 } vmiParameterType;
