61 //
62 // 2 byte endian swap
63 //
64 #define SWAP_2_BYTE(_V) \
65  ({ \
66  Uns16 _R16; \
67  Uns16 _V16 = _V; \
68  _R16 = (((_V16)<<8) | ((_V16)>>8)); \
69  _R16; \
70  })
71 
72 //
73 // 4 byte endian swap
74 //
75 #define SWAP_4_BYTE(_V) \
76  ({ \
77  Uns32 _R32; \
78  Uns32 _V32 = _V; \
79  _R32 = ((SWAP_2_BYTE(_V32)<<16) | (SWAP_2_BYTE(_V32>>16))); \
80  _R32; \
81  })
