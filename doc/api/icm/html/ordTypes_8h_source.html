34 //
35 // Attributes passed to object file loader.
36 //
37 typedef enum ordLoaderAttrsE {
38  ORD_LOAD_DEFAULT = 0x0000, // Default options.
39  ORD_LOAD_PHYSICAL = 0x0001, // Use physical rather than virtual addresses.
40  ORD_LOAD_VERBOSE = 0x0002, // Report each section loaded.
41  ORD_NO_ZERO_BSS = 0x0004, // Do not zero the BSS segment.
42  ORD_SET_START = 0x0008, // Set processor start address to entry address.
43  ORD_LOAD_SYMBOL_ONLY = 0x0010, // Read the symbols but do not modify the memory.
44  ORD_READ_FRAMES = 0x0020, // Read DWARF Information.
45  ORD_ELF_USE_VMA = 0x0040, // Load ELF files using VMA addresses instead of LMA.
46  ORD_LOAD_SECTIONS = 0x0080, // Load ELF sections rather than program headers.
47  ORD_FAIL_IS_ERROR = 0x0100, // If it fails to load, make this an error.
48  ORD_SIGN_EXTEND_ADDR = 0x0200, // Sign extend 32 to 64 bit addresses
49  ORD_PRIVATE_SYMBOLS = 0x0400, // Symbols will loaded into a private table
50  ORD_LIMIT_32 = 0x0800, // Load max 32 bits per write
51  ORD_LIMIT_64 = 0x1000, // Load max 64 bits per write
52  ORD_SET_PROTECTION = 0x2000, // Set protection on memory
53 } ordLoaderAttrs;
